48 void hal_early_init(void)
49 {
50  cli(); /* Disabling interrupts */
51 
52  wdt_enable(HAL_WDT_INTERVAL); /* Setting up watchdog timer */
53 }

55 /* Interrupt handlers */

61 ISR(TIM1_OVF_vect)
62 {
63  /* Processing seconds ticks */
64  hal_timer1_overflows_number --;
65 
66  if (0x00U == hal_timer1_overflows_number)
67  {
68  hal_second_tick();
69  }
70 }
71 
75 ISR(INT0_vect)
76 {
77  hal_tacho_counter ++;
78 }
79 
80 /* ADC measurement completed */
81 ISR(ADC_vect)
82 {
83  /* Adding current measurement */
84  hal_adc_accumulator += ADCL; /* !! MUST read it first !! */
85  hal_adc_accumulator += ((uint16_t)(ADCH & 0b00000011) << 8);
86 
87  /* Averaging completed? */
88  if (hal_adc_avg_counter > 0)
89  {
90  hal_adc_avg_counter --;
91  return;
92  }
93 
94  hal_temperature = hal_adc_accumulator >> HAL_ADC_RESULT_SHIFT_BITS;
95  hal_adc_accumulator = 0x00U;
96  hal_adc_avg_counter = HAL_ADC_NUM_OF_MEASUREMENTS;
97 }

104 void hal_second_tick()
105 {
106  hal_timer1_overflows_number = HAL_TIMER1_OVERFLOWS_PER_SECOND; /* Reload Timer1 overflows counter */
107 
108  /* Update tachometer measurements */
109  hal_tacho = hal_tacho_counter;
110  hal_tacho_counter = 0x00U;
111 
112  /* For mn_seconds_delay() */
113  if (mn_seconds_delay_counter > 0x00U)
114  {
115  mn_seconds_delay_counter --;
116  }
117 
118  /* Raising corresponding flag */
119  hal_flags0 |= _BV(HAL_F0_JUST_TICKED);
120 
121  /* Calling tick function from main code */
122  mn_second_tick();
123 }

128 void hal_init()
129 {
130  /* Setting ports directions */
131  DDRB = HAL_PORTB_DIR;
132 
133  /* Starting cooler (precaution against wrong code) */
134  PORTB = _BV(HAL_COOLER_PIN);
135 
136 
137  /* Setting up Timer0 (PWM source), but do not start cooler
138  * Prescaler 1, fast PWM, non-inverting OC0A output */
139  TCCR0A |= _BV(WGM00) | _BV(WGM01);
140  TCCR0B |= _BV(CS00);
141 
142  /* Set full PWM for case of emergency (i.e. hang on init stage). */
143  hal_set_cooler_duty_cycle(HAL_MAX_DUTY_CYCLE);
144  hal_start_cooler();
145 
146  /* Setting up Timer1 (time source)
147  * Prescaler 16384, interrupt on overflow */
148  TCCR1 |= _BV(CS10) | _BV(CS11) | _BV(CS12) | _BV(CS13);
149  GTCCR |= _BV(PSR1); /* Resetting prescaler */
150  TCNT1 = 0x00; /* And timer too */
151 
152  /* We need to initialize this variable, otherwise first second tick may occur too late*/
153  hal_timer1_overflows_number = HAL_TIMER1_OVERFLOWS_PER_SECOND;
154 
155  TIMSK |= _BV(TOIE1); /* Overflow interrupt */
156 
157  /* Raising edge causes INT0 - tachometer */
158  MCUCR |= _BV(ISC01) | _BV(ISC00);
159  GIMSK |= _BV(INT0);
160 
161  /* Reset tachometer counters */
162  hal_tacho_counter = 0x00U;
163  hal_tacho = 0x00U;
164 
165  /* Setting up ADC */
166  /* For LM35 we need 1.1V reference, for LM335 we need 2.56 */
167  #ifdef USE_LM35_SENSOR
168  /* LM35 - 1.1V reference*/
169  ADMUX = _BV(REFS1);
170  #else
171  /* LM335 - 2.56V reference*/
172  ADMUX = _BV(REFS2) | _BV(REFS1);
173  #endif
174 
175  /* ADC2 as input */
176  ADMUX |= _BV(MUX1);
177  DIDR0 |= _BV(ADC2D); /* No digital input buffer needed on ADC2/PB4 pin */
178 
179  ADCSRB = 0x00; /* Just to be sure that we are in free running mode */
180 
181  /* Enabled, set start flag, enable auto-triggering (according ADCSRB it will be free-running mode),
182  * interrupt on conversion complete, prescaler 128, i.e. clock 8MHz / 128 = 62.5kHz*/
183  ADCSRA = _BV(ADEN) | _BV(ADSC) | _BV(ADATE) | _BV(ADIE) | _BV(ADPS2) | _BV(ADPS1) | _BV(ADPS0);
184 
185  /* Setting ADC-related stuff */
186  hal_adc_accumulator = 0x00U;
187  hal_temperature = 0x00U;
188  hal_adc_avg_counter = HAL_ADC_NUM_OF_MEASUREMENTS;
189 
190  /* Clearing all flags */
191  hal_flags0 = 0x00U;
192 
193  sei(); /* Enabling interrupts, ready to go*/
194 }

203 void hal_set_cooler_duty_cycle(int16_t cycle)
204 {
205  if (cycle > HAL_MAX_DUTY_CYCLE)
206  {
207  cycle = HAL_MAX_DUTY_CYCLE;
208  }
209  else if (cycle < HAL_MIN_DUTY_CYCLE)
210  {
211  cycle = HAL_MIN_DUTY_CYCLE;
212  }
213 
214  OCR0A = (uint8_t)cycle;
215 
216  if (HAL_MIN_DUTY_CYCLE == cycle)
217  {
218  /* Switching cooler off */
219  hal_stop_cooler();
220  }
221  else
222  {
223  /* Switching cooler on */
224  hal_start_cooler();
225  }
226 }
