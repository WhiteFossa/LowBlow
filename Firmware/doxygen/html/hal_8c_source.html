51 }

59 ISR(TIM1_OVF_vect)
60 {
61  /* Processing seconds ticks */
62  hal_timer1_overflows_number --;
63 
64  if (0x00U == hal_timer1_overflows_number)
65  {
66  hal_second_tick();
67  }
68 }
69 
73 ISR(INT0_vect)
74 {
75  hal_tacho_counter ++;
76 }
77 
78 /* ADC measurement completed */
79 ISR(ADC_vect)
80 {
81  /* Adding current measurement */
82  hal_adc_accumulator += ADCL; /* !! MUST read it first !! */
83  hal_adc_accumulator += ((uint16_t)(ADCH & 0b00000011) << 8);
84 
85  /* Averaging completed? */
86  if (hal_adc_avg_counter > 0)
87  {
88  hal_adc_avg_counter --;
89  return;
90  }
91 
92  hal_temperature = hal_adc_accumulator >> HAL_ADC_RESULT_SHIFT_BITS;
93  hal_adc_accumulator = 0x00U;
94  hal_adc_avg_counter = HAL_ADC_NUM_OF_MEASUREMENTS;
95 }

102 void hal_second_tick()
103 {
104  hal_timer1_overflows_number = HAL_TIMER1_OVERFLOWS_PER_SECOND; /* Reload Timer1 overflows counter */
105 
106  /* Update tachometer measurements */
107  hal_tacho = hal_tacho_counter;
108  hal_tacho_counter = 0x00U;
109 
110  /* For mn_seconds_delay() */
111  if (mn_seconds_delay_counter > 0x00U)
112  {
113  mn_seconds_delay_counter --;
114  }
115 
116  /* Raising corresponding flag */
117  hal_flags0 |= _BV(HAL_F0_JUST_TICKED);
118 
119  /* Calling tick function from main code */
120  mn_second_tick();
121 }

126 void hal_init()
127 {
128  /* Setting ports directions */
129  DDRB = HAL_PORTB_DIR;
130 
131  /* Starting cooler (precaution against wrong code) */
132  PORTB = _BV(HAL_COOLER_PIN);
133 
134 
135  /* Setting up Timer0 (PWM source), but do not start cooler
136  * Prescaler 1, fast PWM, non-inverting OC0A output */
137  TCCR0A |= _BV(WGM00) | _BV(WGM01);
138  TCCR0B |= _BV(CS00);
139 
140  /* Set full PWM for case of emergency (i.e. hang on init stage). */
141  hal_set_cooler_duty_cycle(HAL_MAX_DUTY_CYCLE);
142  hal_start_cooler();
143 
144  /* Setting up Timer1 (time source)
145  * Prescaler 16384, interrupt on overflow */
146  TCCR1 |= _BV(CS10) | _BV(CS11) | _BV(CS12) | _BV(CS13);
147  GTCCR |= _BV(PSR1); /* Resetting prescaler */
148  TCNT1 = 0x00; /* And timer too */
149 
150  /* We need to initialize this variable, otherwise first second tick may occur too late*/
151  hal_timer1_overflows_number = HAL_TIMER1_OVERFLOWS_PER_SECOND;
152 
153  TIMSK |= _BV(TOIE1); /* Overflow interrupt */
154 
155  /* Raising edge causes INT0 - tachometer */
156  MCUCR |= _BV(ISC01) | _BV(ISC00);
157  GIMSK |= _BV(INT0);
158 
159  /* Reset tachometer counters */
160  hal_tacho_counter = 0x00U;
161  hal_tacho = 0x00U;
162 
163  /* Setting up ADC */
164  /* For LM35 we need 1.1V reference, for LM335 we need 2.56 */
165  #ifdef USE_LM35_SENSOR
166  /* LM35 - 1.1V reference*/
167  ADMUX = _BV(REFS1);
168  #else
169  /* LM335 - 2.56V reference*/
170  ADMUX = _BV(REFS2) | _BV(REFS1);
171  #endif
172 
173  /* ADC2 as input */
174  ADMUX |= _BV(MUX1);
175  DIDR0 |= _BV(ADC2D); /* No digital input buffer needed on ADC2/PB4 pin */
176 
177  ADCSRB = 0x00; /* Just to be sure that we are in free running mode */
178 
179  /* Enabled, set start flag, enable auto-triggering (according ADCSRB it will be free-running mode),
180  * interrupt on conversion complete, prescaler 128, i.e. clock 8MHz / 128 = 62.5kHz*/
181  ADCSRA = _BV(ADEN) | _BV(ADSC) | _BV(ADATE) | _BV(ADIE) | _BV(ADPS2) | _BV(ADPS1) | _BV(ADPS0);
182 
183  /* Setting ADC-related stuff */
184  hal_adc_accumulator = 0x00U;
185  hal_temperature = 0x00U;
186  hal_adc_avg_counter = HAL_ADC_NUM_OF_MEASUREMENTS;
187 
188  /* Clearing all flags */
189  hal_flags0 = 0x00U;
190 
191  sei(); /* Enabling interrupts, ready to go*/
192 }

201 void hal_set_cooler_duty_cycle(int16_t cycle)
202 {
203  if (cycle > HAL_MAX_DUTY_CYCLE)
204  {
205  cycle = HAL_MAX_DUTY_CYCLE;
206  }
207  else if (cycle < HAL_MIN_DUTY_CYCLE)
208  {
209  cycle = HAL_MIN_DUTY_CYCLE;
210  }
211 
212  OCR0A = (uint8_t)cycle;
213 
214  if (HAL_MIN_DUTY_CYCLE == cycle)
215  {
216  /* Switching cooler off */
217  hal_stop_cooler();
218  }
219  else
220  {
221  /* Switching cooler on */
222  hal_start_cooler();
223  }
224 }
