51 void main()
52 {
53  /* RPM loop disabled at start-up */
54  mn_flags0 = _BV(MN_F0_DISABLE_RPM_LOOP);
55 
56  /* Initialize hardware */
57  hal_init();
58 
59  /* Switching cooler off and waiting for stop */
60  hal_set_cooler_duty_cycle(HAL_MIN_DUTY_CYCLE);
61 
62  hal_tacho = 0xFFU; /* To avoid immediate exit from loop */
63  while(hal_tacho > 0x00U)
64  {
65  mn_seconds_delay(MN_WAIT_FOR_COOLER_STOP);
66  }
67 
68  /* Cooler stopped, slowly increasing duty cycle until MN_COOLER_SPIN_UP_MINIMAL_RPM reached to find
69  * minimal RPM.
70  */
71  while(1)
72  {
73  if ((hal_flags0 & _BV(HAL_F0_JUST_TICKED)) != 0x00U)
74  {
75  /* New measurement arrived */
76  if (hal_tacho < MN_COOLER_SPIN_UP_MINIMAL_RPM)
77  {
78  /* We need more */
79  hal_set_cooler_duty_cycle(hal_get_cooler_duty_cycle() + MN_COOLER_SPIN_UP_DUTY_CYCLE_STEP);
80  }
81  else
82  {
83  break;
84  }
85 
86  hal_flags0 &= ~_BV(HAL_F0_JUST_TICKED); /* Ready for the next measurement */
87  }
88 
89  wdt_reset();
90  }
91 
92  /* Wait and measuring RPM */
93  mn_seconds_delay(MN_COOLER_SPIN_UP_WAIT);
94  mn_min_duty_cycle = hal_get_cooler_duty_cycle();
95  mn_min_rpm = hal_tacho;
96 
97  /* Full throttle and measure again */
98  hal_set_cooler_duty_cycle(HAL_MAX_DUTY_CYCLE);
99 
100  mn_seconds_delay(MN_COOLER_SPIN_UP_WAIT);
101  mn_max_rpm = hal_tacho;
102 
103  /* Do we measured RPM correctly? */
104  if (mn_max_rpm <= mn_min_rpm)
105  {
106  /* Just hang, because we have full throttle set */
107  while(1)
108  {
109  wdt_reset();
110  }
111  }
112 
113  /* Main loop */
114  while(1)
115  {
116  mn_set_rpm_in_percents(mn_lookup_rpm(hal_temperature));
117  wdt_reset();
118  }
119 }

124 void mn_second_tick()
125 {
126  if (0x00 == (mn_flags0 & _BV(MN_F0_DISABLE_RPM_LOOP)))
127  {
128  /* Adjusting RPM */
129  int16_t error = (int16_t)hal_tacho - (int16_t)mn_target_rpm; /* RMP error level */
130 
131  hal_set_cooler_duty_cycle((int16_t)hal_get_cooler_duty_cycle() - error);
132  }
133 }

156 void mn_set_target_rpm(uint16_t target_rpm)
157 {
158  /* Do we need to stop cooler? */
159  if (0x00U == target_rpm)
160  {
161  if (0x00 == (mn_flags0 & _BV(MN_F0_DISABLE_RPM_LOOP)))
162  {
163  /* Yes */
164  hal_set_cooler_duty_cycle(0x00);
165  mn_flags0 |= _BV(MN_F0_DISABLE_RPM_LOOP);
166  }
167  return;
168  }
169 
170  if (target_rpm > mn_max_rpm)
171  {
172  target_rpm = mn_max_rpm;
173  }
174  else
175  {
176  if (target_rpm < mn_min_rpm)
177  {
178  target_rpm = mn_min_rpm;
179  }
180  }
181 
182  mn_target_rpm = target_rpm;
183 
184  /* Do we need to start cooler? */
185  if ((mn_flags0 & _BV(MN_F0_DISABLE_RPM_LOOP)) != 0x00U)
186  {
187  /* Yes */
188  hal_set_cooler_duty_cycle(mn_min_duty_cycle);
189 
190  mn_seconds_delay(MN_WAIT_BEFORE_RPM_LOOP);
191 
192  /* Activating loop */
193  mn_flags0 &= ~_BV(MN_F0_DISABLE_RPM_LOOP);
194  }
195 }

205 void mn_set_rpm_in_percents(uint16_t target_rpm)
206 {
207  uint8_t rpm;
208 
209  if (0 == target_rpm)
210  {
211  rpm = 0;
212  }
213  else
214  {
215  rpm = (((uint32_t)(target_rpm - 1) * (uint32_t)(mn_max_rpm - mn_min_rpm)) >> 8) + mn_min_rpm;
216  }
217 
218  mn_set_target_rpm(rpm);
219 }

225 uint16_t mn_lookup_rpm(uint16_t tmes)
226 {
227  /* Do we exceeded base temperature? */
228  uint16_t baseTemp = eeprom_read_word(MN_LOOKUP_ABL_ADDRESS); /* Temperature (reading base ADC level) */
229 
230  if (tmes < baseTemp)
231  {
232  // Temperature is too low, cooler off.
233  return 0;
234  }
235 
236  uint16_t rpm = eeprom_read_byte(MN_LOOKUP_BRPM_ADDRESS) + MN_BRPM_INCREMENT; /* RPMs at base temperature */
237  uint16_t temp = baseTemp;
238 
239  uint8_t *eeprom_lookup_ptr = (uint8_t*)MN_LOOKUP_FIRST_VALUE_ADDRESS;
240 
241  uint8_t lookup_byte; /* RRRR TTTT */
242  while(tmes > temp)
243  {
244  /* Reading lookup value */
245  lookup_byte = eeprom_read_byte(eeprom_lookup_ptr);
246  eeprom_lookup_ptr ++;
247 
248  if (eeprom_lookup_ptr > E2END)
249  {
250  /* Probably EEPROM unprogrammed */
251  return MN_FULL_THROTTLE;
252  }
253 
254  temp += (lookup_byte & 0b00001111) + MN_LOOKUP_TEMPERATURE_INCREMENT;
255  rpm += (lookup_byte & 0b11110000) >> 4;
256  }
257 
258  return rpm;
259 }
