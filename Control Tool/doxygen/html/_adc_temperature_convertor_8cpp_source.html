37 uint AdcTemperatureConvertor::TEMP2ADC(double temp)
38 {
39  int pre_result = (int)floor((temp - this->b) / this->a + 0.5);
40  if(pre_result < 0)
41  {
42  return 0;
43  }
44  else if (pre_result > ADC_MAX_VALUE)
45  {
46  return ADC_MAX_VALUE;
47  }
48 
49  return (uint)pre_result;
50 }

65 void AdcTemperatureConvertor::LoadSettings(QString filename)
66 {
67  QFile *file = new QFile(filename);
68  if (!file->open(QIODevice::ReadOnly | QIODevice::Text))
69  {
70  SafeDelete(file);
71  throw std::runtime_error(QString(QObject::tr("Cannot open file \"%1\" for reading")).arg(filename).toStdString());
72  }
73 
74  QXmlStreamReader xsr(file);
75 
76  // Reading
77  bool start_found = false; // Load progress flags
78  bool description_loaded = false;
79  bool mul_loaded = false;
80  bool add_loaded = false;
81  while (!xsr.atEnd()) // atEnd() true not only for end of file, but for error too
82  {
83  QXmlStreamReader::TokenType token = xsr.readNext();
84  if (QXmlStreamReader::TokenType::StartElement == token)
85  {
86  if (A2T_SETTINGS_ROOT_ELEMENT == xsr.name())
87  {
88  // Root element
89  QXmlStreamAttributes attrs = xsr.attributes();
90 
91  if (!attrs.hasAttribute(A2T_SETTINGS_DEVICE_ATTR) || !attrs.hasAttribute(A2T_SETTINGS_VERSION_ATTR))
92  {
93  // Missing required attributes
94  file->close();
95  SafeDelete(file);
96  throw std::runtime_error(QString(QObject::tr("Either %1 or %2 attributes missing at %3 element")).arg(A2T_SETTINGS_DEVICE_ATTR).arg(A2T_SETTINGS_VERSION_ATTR).arg(A2T_SETTINGS_ROOT_ELEMENT).toStdString());
97  }
98 
99  if (attrs.value(A2T_SETTINGS_DEVICE_ATTR) != A2T_SETTINGS_DEVICE_NAME)
100  {
101  // Wrong device
102  file->close();
103  SafeDelete(file);
104  throw std::runtime_error(QString(QObject::tr("%1 file is for another device")).arg(filename).toStdString());
105  }
106 
107  if (attrs.value(A2T_SETTINGS_VERSION_ATTR) != A2T_SETTINGS_VERSION)
108  {
109  // Wrong version
110  file->close();
111  SafeDelete(file);
112  throw std::runtime_error(QString(QObject::tr("%1 file have wrong version")).arg(filename).toStdString());
113  }
114 
115  start_found = true;
116  }
117  else if (A2T_SETTINGS_DESCRIPTION_EL == xsr.name())
118  {
119  this->description = xsr.readElementText();
120  description_loaded = true;
121  }
122  else if (A2T_SETTINGS_MULTIPLICATIVE_EL == xsr.name())
123  {
124  bool success = false;
125  this->a = xsr.readElementText().toDouble(&success);
126 
127  if (!success)
128  {
129  file->close();
130  SafeDelete(file);
131  throw std::runtime_error(QString(QObject::tr("Multiplicative parameter in %1 file is not a floating point number")).arg(filename).toStdString());
132  }
133 
134  mul_loaded = true;
135  }
136 
137  else if (A2T_SETTINGS_ADDITIVE_EL == xsr.name())
138  {
139  bool success = false;
140  this->b = xsr.readElementText().toDouble(&success);
141 
142  if (!success)
143  {
144  file->close();
145  SafeDelete(file);
146  throw std::runtime_error(QString(QObject::tr("Additive parameter in %1 file is not a floating point number")).arg(filename).toStdString());
147  }
148 
149  add_loaded = true;
150  }
151  }
152  }
153 
154  // Is it error?
155  if (xsr.hasError())
156  {
157  file->close();
158  SafeDelete(file);
159  throw std::runtime_error(QString(QObject::tr("Error %1 while parsing file %2")).arg(xsr.errorString()).arg(filename).toStdString());
160  }
161 
162  file->close();
163  SafeDelete(file);
164 
165  // Do we have all data?
166  if (!(start_found && description_loaded && mul_loaded && add_loaded))
167  {
168  throw std::runtime_error(QString(QObject::tr("File %1 have missing required records")).arg(filename).toStdString());
169  }
170 }

172 void AdcTemperatureConvertor::SaveSettings(QString filename)
173 {
174  QFile *file = new QFile(filename);
175  if (!file->open(QIODevice::WriteOnly | QIODevice::Truncate | QIODevice::Text))
176  {
177  SafeDelete(file);
178  throw std::runtime_error(QString(QObject::tr("Cannot open file \"%1\" for writing")).arg(filename).toStdString());
179  }
180 
181  QXmlStreamWriter xsw(file); // Stream writer for XML data
182  xsw.setAutoFormatting(true);
183  xsw.writeStartDocument();
184  this->WriteADC2TemperatureSection(&xsw);
185  xsw.writeEndDocument();
186 
187  file->close();
188  SafeDelete(file);
189 }

191 void AdcTemperatureConvertor::WriteADC2TemperatureSection(QXmlStreamWriter *writer)
192 {
193  writer->writeStartElement(A2T_SETTINGS_ROOT_ELEMENT); // Settings root
194  writer->writeAttribute(A2T_SETTINGS_DEVICE_ATTR, A2T_SETTINGS_DEVICE_NAME); // For what device
195  writer->writeAttribute(A2T_SETTINGS_VERSION_ATTR, A2T_SETTINGS_VERSION); // Settings file version
196  writer->writeTextElement(A2T_SETTINGS_DESCRIPTION_EL, this->description); // Settings description
197  writer->writeTextElement(A2T_SETTINGS_MULTIPLICATIVE_EL, FormatDoubleForXML(this->a)); // a
198  writer->writeTextElement(A2T_SETTINGS_ADDITIVE_EL, FormatDoubleForXML(this->b)); // b
199  writer->writeEndElement(); // End of settings root
200 }
