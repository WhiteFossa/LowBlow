48 uint AdcTemperatureConvertor::TEMP2ADC(double temp)
49 {
50  int pre_result = (int)floor((temp - _b) / _a + 0.5);
51  if(pre_result < 0)
52  {
53  return 0;
54  }
55  else if (pre_result > MaxADCValue)
56  {
57  return MaxADCValue;
58  }
59 
60  return (uint)pre_result;
61 }

81 bool AdcTemperatureConvertor::LoadSettings(QString filename, QString prefix)
82 {
83  QFile *file = new QFile(filename);
84  if (!file->open(QIODevice::ReadOnly | QIODevice::Text))
85  {
86  SafeDelete(file);
87  return false;
88  }
89 
90  QXmlQuery query;
91 
92  // Binding settings file contents into $settingsFile variable
93  query.bindVariable("settingsFile", file);
94 
95  // Is it our XML at all?
96  query.setQuery(QString(QObject::tr("%1/%2/@%3/string()")).arg(prefix).arg(SettingsRootElement).arg(DeviceAttribute));
97  QString deviceName = Fossa::Helpers::XmlHelper::GetTextValue(&query);
98 
99  query.setQuery(QString(QObject::tr("%1/%2/@%3/string()")).arg(prefix).arg(SettingsRootElement).arg(VersionAttribute));
100  uint version = Fossa::Helpers::XmlHelper::GetIntegerValue(&query);
101 
102  if ((deviceName != DeviceName) || (version != Version))
103  {
104  file->close();
105  SafeDelete(file);
106  return false;
107  }
108 
109  // Description
110  query.setQuery(QString(QObject::tr("%1/%2/%3/text()")).arg(prefix).arg(SettingsRootElement).arg(DescriptionElement));
111  _description = Fossa::Helpers::XmlHelper::GetTextValue(&query);
112 
113  // Multiplicative coefficient
114  query.setQuery(QString(QObject::tr("%1/%2/%3/text()")).arg(prefix).arg(SettingsRootElement).arg(MultiplicativeElement));
115  _a = Fossa::Helpers::XmlHelper::GetDoubleValue(&query);
116 
117  // Additive coefficient
118  query.setQuery(QString(QObject::tr("%1/%2/%3/text()")).arg(prefix).arg(SettingsRootElement).arg(AdditiveElement));
119  _b = Fossa::Helpers::XmlHelper::GetDoubleValue(&query);
120 
121  file->close();
122  SafeDelete(file);
123 
124  return true;
125 }

127 void AdcTemperatureConvertor::SaveSettings(QString filename)
128 {
129  QFile *file = new QFile(filename);
130  if (!file->open(QIODevice::WriteOnly | QIODevice::Truncate | QIODevice::Text))
131  {
132  SafeDelete(file);
133  throw std::runtime_error(QString(QObject::tr("Cannot open file \"%1\" for writing")).arg(filename).toStdString());
134  }
135 
136  QXmlStreamWriter xsw(file); // Stream writer for XML data
137  xsw.setAutoFormatting(true);
138  xsw.writeStartDocument();
139  WriteADC2TemperatureSection(&xsw);
140  xsw.writeEndDocument();
141 
142  file->close();
143  SafeDelete(file);
144 }

146 void AdcTemperatureConvertor::WriteADC2TemperatureSection(QXmlStreamWriter *writer)
147 {
148  writer->writeStartElement(SettingsRootElement); // Settings root
149  writer->writeAttribute(DeviceAttribute, DeviceName); // For what device
150  writer->writeAttribute(VersionAttribute, QString(QObject::tr("%1")).arg(Version)); // Settings file version
151  writer->writeTextElement(DescriptionElement, _description); // Settings description
152  writer->writeTextElement(MultiplicativeElement, FormatDoubleForXML(_a)); // a
153  writer->writeTextElement(AdditiveElement, FormatDoubleForXML(_b)); // b
154  writer->writeEndElement(); // End of settings root
155 }
