46  return MaxADCValue;

65 bool AdcTemperatureConvertor::LoadSettings(QString filename, QString prefix)
66 {
67  QFile *file = new QFile(filename);
68  if (!file->open(QIODevice::ReadOnly | QIODevice::Text))
69  {
70  SafeDelete(file);
71  return false;
72  }
73 
74  QXmlQuery query;
75 
76  // Binding settings file contents into $settingsFile variable
77  query.bindVariable("settingsFile", file);
78 
79  // Is it our XML at all?
80  query.setQuery(QString(QObject::tr("%1/%2/@%3/string()")).arg(prefix).arg(SettingsRootElement).arg(DeviceAttribute));
81  QString deviceName = Fossa::Helpers::XmlHelper::GetTextValue(&query);
82 
83  query.setQuery(QString(QObject::tr("%1/%2/@%3/string()")).arg(prefix).arg(SettingsRootElement).arg(VersionAttribute));
84  uint version = Fossa::Helpers::XmlHelper::GetIntegerValue(&query);
85 
86  if ((deviceName != DeviceName) || (version != Version))
87  {
88  file->close();
89  SafeDelete(file);
90  return false;
91  }
92 
93  // Description
94  query.setQuery(QString(QObject::tr("%1/%2/%3/text()")).arg(prefix).arg(SettingsRootElement).arg(DescriptionElement));
95  _description = Fossa::Helpers::XmlHelper::GetTextValue(&query);
96 
97  // Multiplicative coefficient
98  query.setQuery(QString(QObject::tr("%1/%2/%3/text()")).arg(prefix).arg(SettingsRootElement).arg(MultiplicativeElement));
99  _a = Fossa::Helpers::XmlHelper::GetDoubleValue(&query);
100 
101  // Additive coefficient
102  query.setQuery(QString(QObject::tr("%1/%2/%3/text()")).arg(prefix).arg(SettingsRootElement).arg(AdditiveElement));
103  _b = Fossa::Helpers::XmlHelper::GetDoubleValue(&query);
104 
105  file->close();
106  SafeDelete(file);
107 
108  return true;
109 }

111 void AdcTemperatureConvertor::SaveSettings(QString filename)
112 {
113  QFile *file = new QFile(filename);
114  if (!file->open(QIODevice::WriteOnly | QIODevice::Truncate | QIODevice::Text))
115  {
116  SafeDelete(file);
117  throw std::runtime_error(QString(QObject::tr("Cannot open file \"%1\" for writing")).arg(filename).toStdString());
118  }
119 
120  QXmlStreamWriter xsw(file); // Stream writer for XML data
121  xsw.setAutoFormatting(true);
122  xsw.writeStartDocument();
123  WriteADC2TemperatureSection(&xsw);
124  xsw.writeEndDocument();
125 
126  file->close();
127  SafeDelete(file);
128 }

130 void AdcTemperatureConvertor::WriteADC2TemperatureSection(QXmlStreamWriter *writer)
131 {
132  writer->writeStartElement(SettingsRootElement); // Settings root
133  writer->writeAttribute(DeviceAttribute, DeviceName); // For what device
134  writer->writeAttribute(VersionAttribute, QString(QObject::tr("%1")).arg(Version)); // Settings file version
135  writer->writeTextElement(DescriptionElement, _description); // Settings description
136  writer->writeTextElement(MultiplicativeElement, FormatDoubleForXML(_a)); // a
137  writer->writeTextElement(AdditiveElement, FormatDoubleForXML(_b)); // b
138  writer->writeEndElement(); // End of settings root
139 }
