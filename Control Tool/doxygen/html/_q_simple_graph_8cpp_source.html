400  QMap<double, double>::const_iterator QSimpleGraph::GetClosestPoint(uint x)
401  {
402  if (0 == this->_points.count())
403  {
404  return NULL;
405  }
406 
407  // X value
408  double xVal = this->GetValByX(x);
409 
410  // Before all
411  if (xVal < this->_points.constBegin().key())
412  {
413  return this->_points.constBegin();
414  }
415 
416  QMap<double, double>::const_iterator prevIterator = this->_points.constBegin();
417  QMap<double, double>::const_iterator nextIterator = prevIterator + 1;
418  while (nextIterator != this->_points.constEnd())
419  {
420  double prevIteratorX = prevIterator.key();
421  double nextIteratorX = nextIterator.key();
422 
423  if ((xVal >= prevIteratorX) && (xVal <= nextIteratorX))
424  {
425  // Who is closer?
426  if ((xVal - prevIteratorX) < (nextIteratorX - xVal))
427  {
428  return prevIterator;
429  }
430  else
431  {
432  return nextIterator;
433  }
434  }
435 
436  // Next point
437  prevIterator = nextIterator;
438  nextIterator ++;
439  }
440  }

452  void QSimpleGraph::DrawTooltip(QPoint targetPoint, TooltipDirection direction, QString text)
453  {
454  QRect textRect = QSimpleGraph::MultilineBoundingRect(this->_painter->font(), text);
455 
456  QRect borderRect = textRect;
457  borderRect.adjust(0, 0, this->_tooltipWidthAdd, this->_tooltipHeightAdd);
458 
459  QPoint shiftedTargetPoint;
460  switch(direction)
461  {
462  case TooltipDirection::DownRight:
463  shiftedTargetPoint.setX(targetPoint.x() + this->_tooltipXShift);
464  shiftedTargetPoint.setY(targetPoint.y() + this->_tooltipYShift);
465  borderRect.moveTopLeft(shiftedTargetPoint);
466  break;
467  case TooltipDirection::DownLeft:
468  shiftedTargetPoint.setX(targetPoint.x() - this->_tooltipXShift);
469  shiftedTargetPoint.setY(targetPoint.y() + this->_tooltipYShift);
470  borderRect.moveTopRight(shiftedTargetPoint);
471  break;
472  case TooltipDirection::UpRight:
473  shiftedTargetPoint.setX(targetPoint.x() + this->_tooltipXShift);
474  shiftedTargetPoint.setY(targetPoint.y() - this->_tooltipYShift);
475  borderRect.moveBottomLeft(shiftedTargetPoint);
476  break;
477  case TooltipDirection::UpLeft:
478  shiftedTargetPoint.setX(targetPoint.x() - this->_tooltipXShift);
479  shiftedTargetPoint.setY(targetPoint.y() - this->_tooltipYShift);
480  borderRect.moveBottomRight(shiftedTargetPoint);
481  break;
482  }
483 
484  this->_painter->save();
485 
486  // Bounding rectangle
487  this->_painter->setPen(this->palette().color(QPalette::ToolTipText));
488  QBrush tooltipBrush(this->palette().color(QPalette::ToolTipBase), Qt::SolidPattern);
489  this->_painter->setBrush(tooltipBrush);
490  this->_painter->drawRect(borderRect);
491 
492  // Drawing text
493  this->_painter->drawText(borderRect, text, Qt::AlignLeft | Qt::AlignVCenter);
494 
495  this->_painter->restore();
496  }

499  {
