163 void MainWindow::MwSlotBaseTemperatureChanged(double tempC)
164 {
165  Interfaces::ISettingsGenerator* setgen = this->_settingsSaverLoader->GetSettingsGeneratorPtr();
166 
167  setgen->SetBaseTemperature(tempC);
168  setgen->CalculateSteps(); // Updating steps values
169 
170  // Setting temperature back (it may not be equal to entered by user due ADC granularity)
171  QObject::disconnect(this->ui->mw_basetemperature, SIGNAL(valueChanged(double)), this, SLOT(MwSlotBaseTemperatureChanged(double))); // Disabling signal-slot connection to avoid recursion
172  this->ui->mw_basetemperature->setValue(setgen->GetBaseTemperature());
173  QObject::connect(this->ui->mw_basetemperature, SIGNAL(valueChanged(double)), this, SLOT(MwSlotBaseTemperatureChanged(double))); // Re-enabling it
174 
175  this->ui->mw_basetemperatureADC->setNum((int)setgen->GetBaseTemperatureADC());
176 
177  // Actualizing UI steps table
178  emit MwSignalUpdateStepsTable();
179 }

189 void MainWindow::InitializeStepsTableColumn(uint col)
190 {
191  QTableWidgetItem* labelItem;
192 
193  // ADC Level (label)
194  labelItem = new QTableWidgetItem(QObject::trUtf8("0"), QTableWidgetItem::Type);
195  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::ADC_LEVEL, col, labelItem);
196 
197  // Temperature (label)
198  labelItem = new QTableWidgetItem(QObject::trUtf8("0.0"), QTableWidgetItem::Type);
199  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::TEMPERATURE, col, labelItem);
200 
201  // RPM level (label)
202  labelItem = new QTableWidgetItem(QObject::trUtf8("0"), QTableWidgetItem::Type);
203  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::RPM_LEVEL, col, labelItem);
204 
205  // RPM percent (label)
206  labelItem = new QTableWidgetItem(QObject::trUtf8("0.0%"), QTableWidgetItem::Type);
207  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::RPM_PERCENT, col, labelItem);
208 
209  // ADC level increase (spinbox)
210  this->_ADCDeltaSpinboxes[col] = new QSpinBox(this->ui->mw_StepsTable);
211  this->ui->mw_StepsTable->setCellWidget(Ui::STEPS_TABLE_ROWS::ADC_DELTA, col, this->_ADCDeltaSpinboxes[col]);
212 
213  // Temperature increase (label)
214  labelItem = new QTableWidgetItem(QObject::trUtf8("0.0"), QTableWidgetItem::Type);
215  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::TEMPERATURE_DELTA, col, labelItem);
216 
217  // RPM increase (spinbox)
218  this->_RPMDeltaSpinboxes[col] = new QSpinBox(this->ui->mw_StepsTable);
219  this->ui->mw_StepsTable->setCellWidget(Ui::STEPS_TABLE_ROWS::RPM_DELTA, col, this->_RPMDeltaSpinboxes[col]);
220 
221  // RPM increase percent (label)
222  labelItem = new QTableWidgetItem(QObject::trUtf8("0.0%"), QTableWidgetItem::Type);
223  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::RPM_DELTA_PERCENT, col, labelItem);
224 
225  if (col < ADDITIONAL_STEPS)
226  {
227  this->_ADCDeltaSpinboxes[col]->setReadOnly(true);
228  this->_RPMDeltaSpinboxes[col]->setReadOnly(true);
229  }
230 
231  // Overriding ranges
232  if (BASE_STEP_INDEX == col)
233  {
234  // Base step, it have wide range
235  this->_ADCDeltaSpinboxes[col]->setRange(0, ADC_MAX_VALUE);
236  this->_RPMDeltaSpinboxes[col]->setRange(0, MAX_RPM);
237  }
238  else
239  {
240  this->_ADCDeltaSpinboxes[col]->setRange(MIN_ADC_DELTA, MAX_ADC_DELTA);
241  this->_RPMDeltaSpinboxes[col]->setRange(0, MAX_RPM_DELTA);
242  }
243 
244  // Connecting here to avoid calling slots during values/ranges initialization
245  QObject::connect(this->_RPMDeltaSpinboxes[col], SIGNAL(valueChanged(int)), this, SLOT(MwSlotRPMDeltaChangedRaw(int)));
246  QObject::connect(this->_ADCDeltaSpinboxes[col], SIGNAL(valueChanged(int)), this, SLOT(MwSlotADCDeltaChangedRaw(int)));
247 }

295 void MainWindow::MwSlotUpdateStepsTable()
296 {
297  Interfaces::ISettingsGenerator* setgen = this->_settingsSaverLoader->GetSettingsGeneratorPtr();
298 
299  // We can do nothing is setgen or this->_graph is not ready yet
300  if ((NULL == setgen) || (NULL == this->_graph))
301  {
302  return;
303  }
304 
305  // Removing old points from graph
306  this->_graph->ClearAllPoints();
307  this->_graph->SetMinXValue(0); // TODO: Fix
308  this->_graph->SetMaxXValue(0);
309  this->_graph->SetMinYValue(0);
310  this->_graph->SetMaxYValue(0);
311 
312  Interfaces::ISettingsStep *stepItem;
313 
314  for (uint step = 0; step < NUMBER_OF_STEPS_TABLE_RECORDS; step++)
315  {
316  // Getting step
317  stepItem = setgen->GetStepPtr(step);
318 
319  // ADC level
320  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::ADC_LEVEL, step)->setText(QString("%1").arg(stepItem->GetCurrentADC()));
321 
322  // Temperature
323  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::TEMPERATURE, step)->setText(QString("%1 C").arg(stepItem->GetCurrentTemperature()));
324 
325  // RPM level
326  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_LEVEL, step)->setText(QString("%1").arg(stepItem->GetCurrentRPM()));
327 
328  // RPM percent
329  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_PERCENT, step)->setText(stepItem->GetCurrentRPMPercentsString());
330 
331  // ADC level increase
332  ((QSpinBox*)this->ui->mw_StepsTable->cellWidget(Ui::STEPS_TABLE_ROWS::ADC_DELTA, step))->setValue(stepItem->GetADCDelta());
333 
334  // Temperature increase
335  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::TEMPERATURE_DELTA, step)->setText(QString("%1 C").arg(stepItem->GetTempDelta()));
336 
337  // RPM level increase
338  ((QSpinBox*)this->ui->mw_StepsTable->cellWidget(Ui::STEPS_TABLE_ROWS::RPM_DELTA, step))->setValue(stepItem->GetRPMDelta());
339 
340  // RPM level increase (percents)
341  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_DELTA_PERCENT, step)->setText(QString("%1%").arg(stepItem->GetRPMDeltaPercents()));
342 
343  // Adding point to graph
344  this->_graph->AddPoint(stepItem->GetCurrentTemperature(), stepItem->GetCurrentRPMPercents());
345  }
346 
347  this->_graph->update();
348 }
