277 void MainWindow::MwSlotUpdateStepsTable()
278 {
279  // We can do nothing is this->_setgen or this->_graph is not ready yet
280  if ((NULL == this->_setgen) || (NULL == this->_graph))
281  {
282  return;
283  }
284 
285  // Removing old points from graph
286  this->_graph->ClearAllPoints();
287  this->_graph->SetMinXValue(0); // TODO: Fix
288  this->_graph->SetMaxXValue(0);
289  this->_graph->SetMinYValue(0);
290  this->_graph->SetMaxYValue(0);
291 
292  Interfaces::ISettingsStep *stepItem;
293 
294  for (uint step = 0; step < NUMBER_OF_STEPS_TABLE_RECORDS; step++)
295  {
296  // Getting step
297  stepItem = this->_setgen->GetStepPtr(step);
298 
299  // ADC level
300  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::ADC_LEVEL, step)->setText(QString("%1").arg(stepItem->GetCurrentADC()));
301 
302  // Temperature
303  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::TEMPERATURE, step)->setText(QString("%1 °C").arg(stepItem->GetCurrentTemperature()));
304 
305  // RPM level
306  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_LEVEL, step)->setText(QString("%1").arg(stepItem->GetCurrentRPM()));
307 
308  // RPM percent
309  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_PERCENT, step)->setText(stepItem->GetCurrentRPMPercentsString());
310 
311  // ADC level increase
312  ((QSpinBox*)this->ui->mw_StepsTable->cellWidget(Ui::STEPS_TABLE_ROWS::ADC_DELTA, step))->setValue(stepItem->GetADCDelta());
313 
314  // Temperature increase
315  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::TEMPERATURE_DELTA, step)->setText(QString("%1 °C").arg(stepItem->GetTempDelta()));
316 
317  // RPM level increase
318  ((QSpinBox*)this->ui->mw_StepsTable->cellWidget(Ui::STEPS_TABLE_ROWS::RPM_DELTA, step))->setValue(stepItem->GetRPMDelta());
319 
320  // RPM level increase (percents)
321  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_DELTA_PERCENT, step)->setText(QString("%1%").arg(stepItem->GetRPMDeltaPercents()));
322 
323  // Adding point to graph
324  this->_graph->AddPoint(stepItem->GetCurrentTemperature(), stepItem->GetCurrentRPMPercents());
325  }
