191 void MainWindow::InitializeStepsTableColumn(uint col)
192 {
193  QTableWidgetItem* labelItem;
194 
195  // ADC Level (label)
196  labelItem = new QTableWidgetItem(QObject::trUtf8("0"), QTableWidgetItem::Type);
197  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::ADC_LEVEL, col, labelItem);
198 
199  // Temperature (label)
200  labelItem = new QTableWidgetItem(QObject::trUtf8("0.0"), QTableWidgetItem::Type);
201  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::TEMPERATURE, col, labelItem);
202 
203  // RPM level (label)
204  labelItem = new QTableWidgetItem(QObject::trUtf8("0"), QTableWidgetItem::Type);
205  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::RPM_LEVEL, col, labelItem);
206 
207  // RPM percent (label)
208  labelItem = new QTableWidgetItem(QObject::trUtf8("0.0%"), QTableWidgetItem::Type);
209  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::RPM_PERCENT, col, labelItem);
210 
211  // ADC level increase (spinbox)
212  this->_ADCDeltaSpinboxes[col] = new QSpinBox(this->ui->mw_StepsTable);
213  this->ui->mw_StepsTable->setCellWidget(Ui::STEPS_TABLE_ROWS::ADC_DELTA, col, this->_ADCDeltaSpinboxes[col]);
214 
215  // Temperature increase (label)
216  labelItem = new QTableWidgetItem(QObject::trUtf8("0.0"), QTableWidgetItem::Type);
217  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::TEMPERATURE_DELTA, col, labelItem);
218 
219  // RPM increase (spinbox)
220  this->_RPMDeltaSpinboxes[col] = new QSpinBox(this->ui->mw_StepsTable);
221  this->ui->mw_StepsTable->setCellWidget(Ui::STEPS_TABLE_ROWS::RPM_DELTA, col, this->_RPMDeltaSpinboxes[col]);
222 
223  // RPM increase percent (label)
224  labelItem = new QTableWidgetItem(QObject::trUtf8("0.0%"), QTableWidgetItem::Type);
225  this->ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::RPM_DELTA_PERCENT, col, labelItem);
226 
227  if (col < ADDITIONAL_STEPS)
228  {
229  this->_ADCDeltaSpinboxes[col]->setReadOnly(true);
230  this->_RPMDeltaSpinboxes[col]->setReadOnly(true);
231  }
232 
233  // Overriding ranges
234  if (BASE_STEP_INDEX == col)
235  {
236  // Base step, it have wide range
237  this->_ADCDeltaSpinboxes[col]->setRange(0, ADC_MAX_VALUE);
238  this->_RPMDeltaSpinboxes[col]->setRange(0, MAX_RPM);
239  }
240  else
241  {
242  this->_ADCDeltaSpinboxes[col]->setRange(MIN_ADC_DELTA, MAX_ADC_DELTA);
243  this->_RPMDeltaSpinboxes[col]->setRange(0, MAX_RPM_DELTA);
244  }
245 
246  // Connecting here to avoid calling slots during values/ranges initialization
247  QObject::connect(this->_RPMDeltaSpinboxes[col], SIGNAL(valueChanged(int)), this, SLOT(MwSlotRPMDeltaChangedRaw(int)));
248  QObject::connect(this->_ADCDeltaSpinboxes[col], SIGNAL(valueChanged(int)), this, SLOT(MwSlotADCDeltaChangedRaw(int)));
249 }

295 void MainWindow::MwSlotUpdateStepsTable()
296 {
297  // We can do nothing is this->_setgen or this->_graph is not ready yet
298  if ((NULL == this->_setgen) || (NULL == this->_graph))
299  {
300  return;
301  }
302 
303  // Removing old points from graph
304  this->_graph->ClearAllPoints();
305  this->_graph->SetMinXValue(0); // TODO: Fix
306  this->_graph->SetMaxXValue(0);
307  this->_graph->SetMinYValue(0);
308  this->_graph->SetMaxYValue(0);
309 
310  Interfaces::ISettingsStep *stepItem;
311 
312  for (uint step = 0; step < NUMBER_OF_STEPS_TABLE_RECORDS; step++)
313  {
314  // Getting step
315  stepItem = this->_setgen->GetStepPtr(step);
316 
317  // ADC level
318  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::ADC_LEVEL, step)->setText(QString("%1").arg(stepItem->GetCurrentADC()));
319 
320  // Temperature
321  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::TEMPERATURE, step)->setText(QString("%1 °C").arg(stepItem->GetCurrentTemperature()));
322 
323  // RPM level
324  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_LEVEL, step)->setText(QString("%1").arg(stepItem->GetCurrentRPM()));
325 
326  // RPM percent
327  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_PERCENT, step)->setText(stepItem->GetCurrentRPMPercentsString());
328 
329  // ADC level increase
330  ((QSpinBox*)this->ui->mw_StepsTable->cellWidget(Ui::STEPS_TABLE_ROWS::ADC_DELTA, step))->setValue(stepItem->GetADCDelta());
331 
332  // Temperature increase
333  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::TEMPERATURE_DELTA, step)->setText(QString("%1 °C").arg(stepItem->GetTempDelta()));
334 
335  // RPM level increase
336  ((QSpinBox*)this->ui->mw_StepsTable->cellWidget(Ui::STEPS_TABLE_ROWS::RPM_DELTA, step))->setValue(stepItem->GetRPMDelta());
337 
338  // RPM level increase (percents)
339  this->ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_DELTA_PERCENT, step)->setText(QString("%1%").arg(stepItem->GetRPMDeltaPercents()));
340 
341  // Adding point to graph
342  this->_graph->AddPoint(stepItem->GetCurrentTemperature(), stepItem->GetCurrentRPMPercents());
343  }
344 
345  this->_graph->update();
346 }
