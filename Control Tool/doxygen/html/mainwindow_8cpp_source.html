218 void MainWindow::InitializeStepsTableColumn(uint col)
219 {
220  QTableWidgetItem* labelItem;
221 
222  // ADC Level (label)
223  labelItem = new QTableWidgetItem(QObject::tr("0"), QTableWidgetItem::Type);
224  ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::ADC_LEVEL, col, labelItem);
225 
226  // Temperature (label)
227  labelItem = new QTableWidgetItem(QObject::tr("0.0"), QTableWidgetItem::Type);
228  ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::TEMPERATURE, col, labelItem);
229 
230  // RPM level (label)
231  labelItem = new QTableWidgetItem(QObject::tr("0"), QTableWidgetItem::Type);
232  ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::RPM_LEVEL, col, labelItem);
233 
234  // RPM percent (label)
235  labelItem = new QTableWidgetItem(QObject::tr("0.0%"), QTableWidgetItem::Type);
236  ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::RPM_PERCENT, col, labelItem);
237 
238  // ADC level increase (spinbox)
239  _ADCDeltaSpinboxes[col] = new QSpinBox(ui->mw_StepsTable);
240  ui->mw_StepsTable->setCellWidget(Ui::STEPS_TABLE_ROWS::ADC_DELTA, col, _ADCDeltaSpinboxes[col]);
241 
242  // Temperature increase (label)
243  labelItem = new QTableWidgetItem(QObject::tr("0.0"), QTableWidgetItem::Type);
244  ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::TEMPERATURE_DELTA, col, labelItem);
245 
246  // RPM increase (spinbox)
247  _RPMDeltaSpinboxes[col] = new QSpinBox(ui->mw_StepsTable);
248  ui->mw_StepsTable->setCellWidget(Ui::STEPS_TABLE_ROWS::RPM_DELTA, col, _RPMDeltaSpinboxes[col]);
249 
250  // RPM increase percent (label)
251  labelItem = new QTableWidgetItem(QObject::tr("0.0%"), QTableWidgetItem::Type);
252  ui->mw_StepsTable->setItem(Ui::STEPS_TABLE_ROWS::RPM_DELTA_PERCENT, col, labelItem);
253 
254  if (col < Interfaces::ISettingsGenerator::AdditionalSteps)
255  {
256  _ADCDeltaSpinboxes[col]->setReadOnly(true);
257  _RPMDeltaSpinboxes[col]->setReadOnly(true);
258  }
259 
260  // Overriding ranges
261  if (SettingsGenerator::BaseLevelsStepIndex == col)
262  {
263  // Base step, it have wide range
264  _ADCDeltaSpinboxes[col]->setRange(0, AdcTemperatureConvertor::MaxADCValue);
265  _RPMDeltaSpinboxes[col]->setRange(0, MAX_RPM);
266  }
267  else
268  {
269  _ADCDeltaSpinboxes[col]->setRange(MIN_ADC_DELTA, MAX_ADC_DELTA);
270  _RPMDeltaSpinboxes[col]->setRange(0, MAX_RPM_DELTA);
271  }
272 
273  // Connecting here to avoid calling slots during values/ranges initialization
274  QObject::connect(_RPMDeltaSpinboxes[col], SIGNAL(valueChanged(int)), this, SLOT(MwSlotRPMDeltaChangedRaw(int)));
275  QObject::connect(_ADCDeltaSpinboxes[col], SIGNAL(valueChanged(int)), this, SLOT(MwSlotADCDeltaChangedRaw(int)));
276 }

330 void MainWindow::MwSlotUpdateStepsTable()
331 {
332  Interfaces::ISettingsGenerator* setgen = _settingsSaverLoader->GetSettingsGeneratorPtr();
333 
334  // We can do nothing is setgen or _graph is not ready yet
335  if ((NULL == setgen) || (NULL == _graph))
336  {
337  return;
338  }
339 
340  // Removing old points from graph
341  _graph->ClearAllPoints();
342  _graph->SetMinXValue(0); // TODO: Fix
343  _graph->SetMaxXValue(0);
344  _graph->SetMinYValue(0);
345  _graph->SetMaxYValue(0);
346 
347  Interfaces::ISettingsStep *stepItem;
348 
349  for (uint step = 0; step < Interfaces::ISettingsGenerator::TotalSteps; step++)
350  {
351  // Getting step
352  stepItem = setgen->GetStepPtr(step);
353 
354  // ADC level
355  ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::ADC_LEVEL, step)->setText(QString("%1").arg(stepItem->GetCurrentADC()));
356 
357  // Temperature
358  ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::TEMPERATURE, step)->setText(QString("%1 °C").arg(stepItem->GetCurrentTemperature()));
359 
360  // RPM level
361  ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_LEVEL, step)->setText(QString("%1").arg(stepItem->GetCurrentRPM()));
362 
363  // RPM percent
364  ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_PERCENT, step)->setText(stepItem->GetCurrentRPMPercentsString());
365 
366  // ADC level increase
367  ((QSpinBox*)ui->mw_StepsTable->cellWidget(Ui::STEPS_TABLE_ROWS::ADC_DELTA, step))->setValue(stepItem->GetADCDelta());
368 
369  // Temperature increase
370  ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::TEMPERATURE_DELTA, step)->setText(QString("%1 °C").arg(stepItem->GetTempDelta()));
371 
372  // RPM level increase
373  ((QSpinBox*)ui->mw_StepsTable->cellWidget(Ui::STEPS_TABLE_ROWS::RPM_DELTA, step))->setValue(stepItem->GetRPMDelta());
374 
375  // RPM level increase (percents)
376  ui->mw_StepsTable->item(Ui::STEPS_TABLE_ROWS::RPM_DELTA_PERCENT, step)->setText(QString("%1%").arg(stepItem->GetRPMDeltaPercents()));
377 
378  // Adding point to graph
379  _graph->AddPoint(stepItem->GetCurrentTemperature(), stepItem->GetCurrentRPMPercents());
380  }
381 
382  _graph->update();
383 }
