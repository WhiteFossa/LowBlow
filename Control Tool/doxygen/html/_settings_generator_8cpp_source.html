93 void SettingsGenerator::CalculateSteps()
94 {
95  // Zero levels step
96  Interfaces::ISettingsStep *stepPtr = _steps[ZeroLevelsStepIndex];
97 
98  stepPtr->SetADCDelta(0);
99  stepPtr->SetRPMDelta(0);
100  stepPtr->CalculatePrimaryValues(0, 0);
101 
102  // Base levels step
103  stepPtr = _steps[BaseLevelsStepIndex];
104  stepPtr->SetADCDelta(_baseADC, true);
105  stepPtr->SetRPMDelta(_baseRPM, true);
106 
107  // Iterating through steps (except first)
108  for (int i = 1; i < _steps.size(); i++)
109  {
110  // Previous element
111  Interfaces::ISettingsStep* prev = _steps[i - 1];
112  _steps[i]->CalculatePrimaryValues(prev->GetCurrentADC(), prev->GetCurrentRPM());
113  }
114 }
