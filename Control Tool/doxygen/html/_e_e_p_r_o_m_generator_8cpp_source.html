22 QByteArray EEPROMGenerator::GetEEPROMContents(Interfaces::ISettingsGenerator* setGenPtr)
23 {
24  uint eepromSize = BaseLevelsSize + Interfaces::ISettingsGenerator::StepsNumber;
25 
26  QByteArray result;
27 
28  result.resize(eepromSize);
29 
30  // Base temperature
31  uint16_t baseTemperature = (uint16_t)setGenPtr->GetBaseTemperatureADC();
32  result[BaseTemperatureLSBAddr] = baseTemperature & BaseTemperatureLSBMask;
33  result[BaseTemperatureMSBAddr] = (baseTemperature & BaseTemperatureMSBMask) >> BaseTemperatureMSBRShift;
34 
35  // Base RPMs
36  result[BaseRPMsAddr] = (uint8_t)((int)setGenPtr->GetBaseRPM() + BaseRPMsShift);
37 
38  // Steps
39  for (uint stepIndex = 0; stepIndex < Interfaces::ISettingsGenerator::StepsNumber; stepIndex ++)
40  {
41  auto step = setGenPtr->GetStepPtrRelative(stepIndex);
42 
43  uint8_t adcDelta = (uint8_t)((int)step->GetADCDelta() + ADCDeltaShift) & HalfByteMaskL;
44  uint8_t RPMsDelta = (uint8_t)step->GetRPMDelta() & HalfByteMaskL;
45 
46  result[stepIndex + BaseLevelsSize] = (uint8_t)((RPMsDelta << RPMsDeltaLShift) | adcDelta);
47  }
48 
49  return result;
50 }
