62 bool SettingsSaverLoader::Load(QString path)
63 {
64  _path = path;
65  _isModified = false;
66 
67  // Loading sensor settings
68  _adc2Temp->LoadSettings(_path, Adc2TempSettingsPrefixFromSettingsFile);
69 
70  SafeDelete(_setgen);
71  _setgen = new SettingsGenerator(_adc2Temp);
72 
73  QFile *file = new QFile(_path);
74  if (!file->open(QIODevice::ReadOnly | QIODevice::Text))
75  {
76  SafeDelete(file);
77  throw std::runtime_error(QString(QObject::tr("Cannot open file \"%1\" for reading.")).arg(_path).toStdString());
78  }
79 
80  QXmlQuery query;
81 
82  // Binding settings file contents into $settingsFile variable
83  query.bindVariable("settingsFile", file);
84 
85  // Is it our XML at all?
86  query.setQuery(QString(QObject::tr("doc($settingsFile)/%1/@%2/string()")).arg(SettingsRootElement).arg(SettingsDeviceAttribute));
87  QString deviceName = Fossa::Helpers::XmlHelper::GetTextValue(&query);
88 
89  // Version
90  query.setQuery(QString(QObject::tr("doc($settingsFile)/%1/@%2/string()")).arg(SettingsRootElement).arg(SettingsVersionAttribute));
91  uint version = Fossa::Helpers::XmlHelper::GetIntegerValue(&query);
92 
93  if ((deviceName != SettingsDeviceName) || (version != SettingsVersion))
94  {
95  // Wrong file
96  file->close();
97  SafeDelete(file);
98 
99  return false;
100  }
101 
102  // Base levels
103  QString baseLevelsXPath = QString(QObject::tr("doc($settingsFile)/%1/%2/")).arg(SettingsRootElement).arg(BaseLevelsElement);
104 
105  // ADC level
106  query.setQuery(QString(QObject::tr("%1TemperatureADC/text()")).arg(baseLevelsXPath));
107  _setgen->SetBaseTemperatureADC(Fossa::Helpers::XmlHelper::GetIntegerValue(&query));
108 
109  // Base RPMs PWM level
110  query.setQuery(QString(QObject::tr("%1RPMsPWM/text()")).arg(baseLevelsXPath));
111  _setgen->SetBaseRPM(Fossa::Helpers::XmlHelper::GetIntegerValue(&query));
112 
113  // Steps
114  QString stepsXPath = QString(QObject::tr("doc($settingsFile)/%1/%2/")).arg(SettingsRootElement).arg(StepsElement);
115 
116  for (uint step = 0; step < Interfaces::ISettingsGenerator::StepsNumber; step ++)
117  {
118  Interfaces::ISettingsStep* stepData = _setgen->GetStepPtrRelative(step);
119 
120  QString xmlStepName = QString(QObject::tr("%1Step%2/")).arg(stepsXPath).arg(step);
121 
122  // ADC increase for step
123  query.setQuery(QString(QObject::tr("%1%2/text()")).arg(xmlStepName).arg(ADCIncreaseElement));
124  stepData->SetADCDelta(Fossa::Helpers::XmlHelper::GetIntegerValue(&query));
125 
126  // RPM increase for step
127  query.setQuery(QString(QObject::tr("%1%2/text()")).arg(xmlStepName).arg(RPMsPWMIncreaseElement));
128  stepData->SetRPMDelta(Fossa::Helpers::XmlHelper::GetIntegerValue(&query));
129  }
130 
131  file->close();
132  SafeDelete(file);
133 
134  return true;
135 }

137 void SettingsSaverLoader::SaveAtGivenPath(QString path)
138 {
139  if (path.isEmpty())
140  {
141  throw std::runtime_error(QObject::tr("Attempt to save file while path is not specified.").toStdString());
142  }
143 
144  // Saving
145  QFile *file = new QFile(path);
146  if (!file->open(QIODevice::WriteOnly | QIODevice::Truncate | QIODevice::Text))
147  {
148  SafeDelete(file);
149  throw std::runtime_error(QString(QObject::tr("Cannot open file \"%1\" for writing.")).arg(path).toStdString());
150  }
151 
152  QXmlStreamWriter xsw(file); // Stream writer for XML data
153  xsw.setAutoFormatting(true);
154  xsw.writeStartDocument();
155  // Writing version and device
156  xsw.writeStartElement(SettingsRootElement); // Settings root
157 
158  xsw.writeAttribute(SettingsDeviceAttribute, SettingsDeviceName); // Device
159  xsw.writeAttribute(SettingsVersionAttribute, QString::number(SettingsVersion)); // Settings version
160 
161  // Writing ADC->Temperature settings
162  _adc2Temp->WriteADC2TemperatureSection(&xsw);
163 
164  // Base levels
165  xsw.writeStartElement(BaseLevelsElement);
166  xsw.writeTextElement(BaseADCLevelElement, QString::number(_setgen->GetBaseTemperatureADC())); // Temperature
167  xsw.writeTextElement(BaseRPMsLevelElement, QString::number(_setgen->GetBaseRPM())); // RPMs
168  xsw.writeEndElement();
169 
170  // Steps
171  xsw.writeStartElement(StepsElement);
172  for (uint step = 0; step < Interfaces::ISettingsGenerator::StepsNumber; step ++)
173  {
174  xsw.writeStartElement(StepElement.arg(step));
175  Interfaces::ISettingsStep* stepPtr = _setgen->GetStepPtrRelative(step);
176  xsw.writeTextElement(ADCIncreaseElement, QString::number(stepPtr->GetADCDelta()));
177  xsw.writeTextElement(RPMsPWMIncreaseElement, QString::number(stepPtr->GetRPMDelta()));
178  xsw.writeEndElement();
179  }
180  xsw.writeEndElement();
181 
182  xsw.writeEndElement();
183 
184  xsw.writeEndDocument();
185 
186  file->close();
187  SafeDelete(file);
188 }

237 void SettingsSaverLoader::ExportToEEPROM(QString path)
238 {
239  auto eepromContents = _eepromGen->GetEEPROMContents(_setgen);
240 
241  EEPROMBuffer.clear();
242 
243  // Generating iHEX
244  struct ihex_state ihex;
245  ihex_init(&ihex);
246  ihex_write_at_address(&ihex, 0); // Start address
247  ihex_write_bytes(&ihex, eepromContents.constData(), eepromContents.count());
248  ihex_end_write(&ihex);
249 
250  // Writing to file
251  QFile *file = new QFile(path);
252  if (!file->open(QIODevice::WriteOnly | QIODevice::Truncate | QIODevice::Text))
253  {
254  SafeDelete(file);
255  throw std::runtime_error(QString(QObject::tr("Cannot open file \"%1\" for writing.")).arg(path).toStdString());
256  }
257 
258  if (file->write(EEPROMBuffer) != EEPROMBuffer.count())
259  {
260  file->close();
261  SafeDelete(file);
262  throw std::runtime_error(QString(QObject::tr("Error during write to \"%1\".")).arg(path).toStdString());
263  }
264  file->close();
265  SafeDelete(file);
266 }
